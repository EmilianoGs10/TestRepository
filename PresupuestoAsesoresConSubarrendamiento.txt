SELECT         CONVERT(VARCHAR(6), CONVERT(date, dbo.Fecha(T0.FECHA)), 112) as Periodo, CASE WHEN LEFT(T0.MONEDA, 1) = 'P' THEN 1 ELSE T0.TIPOCAMBIO END * T1.IMPORTE AS SUBTOTALPESOS, T0.IDEMPLEADO,  
                         T1.MTIPO, T0.IDSUCURSAL, er.IDEQUIPO, er.ARRENDADORA AS SUBARRENDADO, T0.IDFACTURA --, T0.IDFACTURA
FROM            dbo.OperFacturas T0 
				inner join dbo.OperConFac T1 ON T0.IDFACTURA = T1.FACTURASNUMERO
				left outer join CataEquiposRenta AS er ON T1.IDEQUIPORENTA = er.IDEQUIPO
WHERE        (YEAR(dbo.Fecha(T0.FECHA)) >= '2019') AND (T0.CANCELADA = 'N') --AND T0.IDFACTURA = 381068
GROUP BY CONVERT(VARCHAR(6), CONVERT(date, dbo.Fecha(T0.FECHA)), 112),  T0.IDEMPLEADO, T1.MTIPO, T0.IDSUCURSAL,T0.MONEDA, T0.TIPOCAMBIO, T1.IMPORTE, er.IDEQUIPO, er.ARRENDADORA, T0.IDFACTURA --,T0.IDFACTURA
ORDER BY CONVERT(VARCHAR(6), CONVERT(date, dbo.Fecha(T0.FECHA)), 112)
